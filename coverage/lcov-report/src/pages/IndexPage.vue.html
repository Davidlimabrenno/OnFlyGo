<script setup lang="ts">
import { ref, reactive, onMounted, computed, watch } from 'vue'
import type { QInfiniteScroll } from 'quasar'
import {
  QPage,
  QIcon,
  QBtn,
  QSelect,
  QImg,
  QRating,
  QDrawer,
  QCarousel,
  QCarouselSlide,
} from 'quasar'
 
import placesData from 'src/assets/place.json'
import hotelsData from 'src/assets/hotel.json'
 
interface Hotel {
  id: number
  name: string
  stars: string
  price: number
  images: string[]
  amenities: { key: string; label: string }[] | null
  hasRefundableRoom: boolean
  address: { fullAddress: string }
  placeId: number
  description?: string
}
 
interface Place {
  name: string
  state: { name: string; shortname: string }
  placeId: number
}

interface DestinoOption {
  fullName: string
  shortName: string
  placeId: number
}

interface HotelGroupFromJson {
  placeId: number
  hotels: Omit<Hotel, 'placeId'>[]
}
 
// HOTÉIS
const hotels = ref<Hotel[]>([])
onMounted(() => {
  const groups = hotelsData as HotelGroupFromJson[]
  hotels.value = groups.flatMap((g) => g.hotels.map((h) => ({ ...h, placeId: g.placeId })))
  // Inicializa activeImageMap para cada hotel
  hotels.value.forEach((hotel) => {
    activeImageMap[hotel.id] = 0
  })
})

// DESTINOS
const selectedPlaceId = ref<number | null>(null)
const appliedPlaceId = ref<number | null>(null)

const destinationsOptions = ref<DestinoOption[]>([])
onMounted(() => {
  const arrPlaces = placesData as Place[]
  destinationsOptions.value = arrPlaces.map((pl) => ({
    fullName: `${pl.name}, ${pl.state.name}`,
    shortName: `${pl.name}, ${pl.state.shortname}`,
    placeId: pl.placeId,
  }))
})

// FILTROS
const hasSearched = ref(false)
const sortOrder = ref<'recommended' | 'bestRated'>('recommended')
const sortOptions = [
  { label: 'Recomendados', value: 'recommended' },
  { label: 'Melhor Avaliados', value: 'bestRated' },
]
const currentLimit = ref(10)
 
// FILTRO DESTINO
const filteredHotels = computed(() => {
  return appliedPlaceId.value
    ? hotels.value.filter((h) => h.placeId === appliedPlaceId.value)
    : hotels.value
})

// ORDENAR
const sortedHotels = computed(() => {
  const arr = [...filteredHotels.value]
  if (sortOrder.value === 'recommended') {
    arr.sort((a, b) => a.price - b.price)
  } else {
    arr.sort((a, b) => Number(b.stars) - Number(a.stars))
  }
  return arr
})
 
// EXIBIR
const displayedHotels = computed(() => sortedHotels.value.slice(0, currentLimit.value))
 
watch([appliedPlaceId, sortOrder], () => {
  currentLimit.value = 10
})
 
const breadcrumbCity = computed(() => {
  if (appliedPlaceId.value) {
    const place = (placesData as Place[]).find((p) => p.placeId === appliedPlaceId.value)
    return place ? `${place.name}, ${place.state.shortname}` : ''
  }
  return ''
})
 
// SCROLL
function loadMore(index: number, done: (stop?: boolean) => void) {
  currentLimit.value += 10
  done()
}
 
function onSearch() {
  appliedPlaceId.value = selectedPlaceId.value
 
  hasSearched.value = true
  currentLimit.value = 10
}
 
// AUTOCOMPLETE
function filterPlaces(val: string, update: (fn: () => void) => void) {
  update(() => {
    const arrPlaces = placesData as Place[]
    if (!val) {
      destinationsOptions.value = arrPlaces.map((pl) => ({
        fullName: `${pl.name}, ${pl.state.name}`,
        shortName: `${pl.name}, ${pl.state.shortname}`,
        placeId: pl.placeId,
      }))
      return
    }
    const lowerVal = val.toLowerCase()
    destinationsOptions.value = arrPlaces
      .filter((pl) => `${pl.name}, ${pl.state.name}`.toLowerCase().includes(lowerVal))
      .map((pl) => ({
        fullName: `${pl.name}, ${pl.state.name}`,
        shortName: `${pl.name}, ${pl.state.shortname}`,
        placeId: pl.placeId,
      }))
  })
}
 
function onClearDestino() {
  appliedPlaceId.value = null
}
 
// AUXILIARES
function formatPrice(value: number) {
  return value.toLocaleString('pt-BR', { minimumFractionDigits: 2 })
}
 
// DRAWER
const showDrawer = ref(false)
const drawerWidth = ref(700)
const selectedHotel = ref<Hotel | null>(null)
const activeImage = ref(0)
const showAllAmenities = ref(false)
const amenitiesToShow = computed(() => {
  if (!selectedHotel.value?.amenities) return []
  return showAllAmenities.value
    ? selectedHotel.value.amenities
    : selectedHotel.value.amenities.slice(0, 4)
})
 
function showDetails(hotel: Hotel) {
  selectedHotel.value = hotel
  activeImage.value = 0
  showAllAmenities.value = false
  showDrawer.value = true
}
 
function onClickOutside() {
  showDrawer.value = false
}
 
function toggleAmenities() {
  showAllAmenities.value = !showAllAmenities.value
}
 
function amenityIcon(key: string) {
  const dict: Record<string, string> = {
    WI_FI: 'wifi',
    BREAKFAST: 'free_breakfast',
    PARKING: 'directions_car',
    RESTAURANT: 'restaurant',
    LAUNDRY: 'local_laundry_service',
    POOL: 'pool',
    PETS: 'pets',
    AIR_CONDITIONING: 'ac_unit',
    ROOM_SERVICE: 'room_service',
    FITNESS_CENTER: 'fitness_center',
  }
  return dict[key] || 'done'
}
 
// Objeto reativo que guarda o índice do slide ativo para cada hotel
const activeImageMap = reactive<Record<number, number>>({})
 
//Fim do script
defineExpose({
  showDetails,
  onClickOutside,
  showDrawer,
  hasSearched,
})
</script>
